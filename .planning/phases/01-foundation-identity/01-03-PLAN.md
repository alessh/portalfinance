       export async function sendEmail(params: { to: string; subject: string; template: ReactElement }): Promise<{ messageId: string | null; suppressed: boolean }> {
         const to_lower = params.to.toLowerCase();
         const [suppressed] = await db.select().from(ses_suppressions).where(eq(ses_suppressions.email_lower, to_lower));
         if (suppressed) return { messageId: null, suppressed: true };

         const html = await render(params.template);
         const cmd = new SendEmailCommand({
           Source: env.SES_FROM_EMAIL ?? 'no-reply@portalfinance.app',
           Destination: { ToAddresses: [params.to] },
           Message: {
             Subject: { Data: params.subject, Charset: 'UTF-8' },
             Body: { Html: { Data: html, Charset: 'UTF-8' } },
           },
         });
         const res = await ses.send(cmd);
         return { messageId: res.MessageId ?? null, suppressed: false };
       }
