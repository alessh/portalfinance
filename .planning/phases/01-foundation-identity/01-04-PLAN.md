    3. `src/lib/sentry.ts` — the shared beforeSend + user-id hasher (RESEARCH.md § Plan slice 01-04 item 1). Synchronous (RESEARCH.md Pitfall 5):
       ```ts
       import type { Event } from '@sentry/types';
       import { createHash } from 'node:crypto';
       import { scrubString, scrubObject } from '@/lib/piiScrubber';
       import { env } from '@/lib/env';

       export function hashUserIdForSentry(user_id: string): string {
         // Deterministic-but-private hash; server secret prevents cross-project correlation
         return createHash('sha256').update(env.NEXTAUTH_SECRET).update(user_id).digest('hex').slice(0, 16);
       }

       export function beforeSend(event: Event): Event | null {
         try {
           if (event.message) event.message = scrubString(event.message);
           event.exception?.values?.forEach((ex) => { if (ex.value) ex.value = scrubString(ex.value); });
           event.breadcrumbs?.forEach((bc) => { if (bc.message) bc.message = scrubString(bc.message); });
           if (event.extra) event.extra = scrubObject(event.extra) as typeof event.extra;
           if (event.contexts) event.contexts = scrubObject(event.contexts) as typeof event.contexts;
           if (event.user?.id) event.user = { id: hashUserIdForSentry(String(event.user.id)) };
           else if (event.user) event.user = undefined; // drop any email / ip from user
         } catch {
           // NEVER throw from beforeSend per Sentry contract (Pitfall 5)
         }
         return event;
       }
       ```
       `node:crypto` import is OK here because this file is imported ONLY from `sentry.server.config.ts` and `sentry.edge.config.ts` — edge does not have node:crypto, so provide an edge-safe alternative:

       Alternative: split into `src/lib/sentry.ts` (shared scrub logic, no node:crypto) + `src/lib/sentryServer.ts` (adds the hash function). The edge beforeSend uses `event.user = undefined` instead of hashing IDs (edge traffic is typically middleware — user IDs aren't attached yet). Document this trade-off in a comment.

       Decision for this plan: keep a single `src/lib/sentry.ts` with `hashUserIdForSentry` gated by `typeof process !== 'undefined' && process.versions?.node`. On edge, `event.user` is set to `undefined` entirely.

       ```ts
       export const runtime = 'nodejs';

       import { db } from '@/db';
       import { webhook_events } from '@/db/schema';
       import { enqueue, QUEUES } from '@/jobs/boss';
       import { verifySnsMessage, type SnsMessage } from '@/lib/snsVerifier';
       import { logger } from '@/lib/logger';

       export async function POST(req: Request) {
         const start = Date.now();
         let body: SnsMessage;
         try {
           body = (await req.json()) as SnsMessage;
         } catch {
           return new Response('bad json', { status: 400 });
         }

         // 1. Verify signature (blocking: no suppression poisoning)
         const ok = await verifySnsMessage(body);
         if (!ok) {
           logger.warn({ event: 'ses_bounce_signature_failed', messageId: body.MessageId }, 'SNS signature rejected');
           return new Response('bad signature', { status: 401 });
         }

         // 2. Handle subscription confirmation (one-time dance)
         if (body.Type === 'SubscriptionConfirmation' && body.SubscribeURL) {
           // Confirm by fetching SubscribeURL
           await fetch(body.SubscribeURL).catch((err) => logger.error({ err }, 'subscribe confirm failed'));
           return new Response('subscribed', { status: 200 });
         }

         // 3. Idempotent insert (UNIQUE(source, event_id))
         if (body.Type === 'Notification') {
           const inserted = await db
             .insert(webhook_events)
             .values({
               source: 'SES',
               event_type: 'bounce',
               event_id: body.MessageId,
               payload: body as any,
             })
             .onConflictDoNothing({ target: [webhook_events.source, webhook_events.event_id] })
             .returning({ id: webhook_events.id });

           if (inserted.length > 0) {
             await enqueue(QUEUES.SES_BOUNCE, { webhook_event_id: inserted[0].id });
           }
           logger.info({ event: 'ses_bounce_received', messageId: body.MessageId, latency_ms: Date.now() - start, was_duplicate: inserted.length === 0 });
         }

         return new Response('ok', { status: 200 });
       }
       ```

    4. `src/jobs/workers/sesBounceWorker.ts`:
       ```ts
       import type { Job } from 'pg-boss';
       import { db } from '@/db';
       import { webhook_events, ses_suppressions } from '@/db/schema';
       import { eq, sql } from 'drizzle-orm';
       import { logger } from '@/lib/logger';

       interface Payload { webhook_event_id: string }

       interface SesBounceNotification {
         notificationType: 'Bounce' | 'Complaint';
         mail: { messageId: string };
         bounce?: {
           bouncedRecipients: Array<{ emailAddress: string }>;
           bounceType?: string;
         };
         complaint?: {
           complainedRecipients: Array<{ emailAddress: string }>;
         };
       }

       export async function sesBounceWorker(jobs: Job<Payload>[]) {
         for (const job of jobs) {
           const [ev] = await db.select().from(webhook_events).where(eq(webhook_events.id, job.data.webhook_event_id));
           if (!ev || ev.processed_at) continue;

           // payload.Message is the JSON-encoded SES notification (SNS envelope)
           const envelope = ev.payload as any;
           let notif: SesBounceNotification;
           try {
             notif = JSON.parse(envelope.Message ?? '{}');
           } catch {
             logger.error({ event: 'ses_bounce_malformed', id: ev.id });
             continue;
           }

           const recipients =
             notif.bounce?.bouncedRecipients?.map((r) => r.emailAddress)
             ?? notif.complaint?.complainedRecipients?.map((r) => r.emailAddress)
             ?? [];
           const reason = notif.notificationType === 'Complaint' ? 'COMPLAINT' : 'BOUNCE';

           for (const email of recipients) {
             await db
               .insert(ses_suppressions)
               .values({
                 email_lower: email.toLowerCase(),
                 reason,
                 notification_ids: [envelope.MessageId],
               })
               .onConflictDoUpdate({
                 target: ses_suppressions.email_lower,
                 set: {
                   notification_ids: sql`array_append(${ses_suppressions.notification_ids}, ${envelope.MessageId})`,
                 },
               });
           }

           await db.update(webhook_events).set({ processed_at: new Date() }).where(eq(webhook_events.id, ev.id));
         }
       }
       ```
