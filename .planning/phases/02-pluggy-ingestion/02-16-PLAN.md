import { describe, it, expect, beforeAll, afterAll } from 'vitest';
import { getBoss, QUEUES } from '@/jobs/boss';
import type { Job } from 'pg-boss';
// + project's testcontainers helpers (mirror tests/integration/pluggy/*.test.ts setup; the
//   singleton testcontainer from plan 02-09 is already booted via globalSetup)

describe('pg-boss singleton semantics (Concern #8)', () => {
  beforeAll(async () => {
    // pg-boss is started by globalSetup (plan 02-09); we just need the handle.
    // No additional setup unless test mode shuts it down between suites.
  });

  it('pgboss-singleton-1: 5 enqueues with same singletonKey + singletonHours=0 do not produce 5 completions (in-flight + minimal backlog dedup)', async () => {
    const boss = await getBoss();
    let completed = 0;
    const completionTimes: number[] = [];

    // Register a slow handler. Use a unique queue name so we don't collide with
    // the production worker registration.
    const test_queue = `pluggy.sync.test.${Date.now()}`;
    await boss.createQueue(test_queue);
    await boss.work(test_queue, { localConcurrency: 1 }, async (jobs: Job<unknown>[]) => {
      for (const _ of jobs) {
        await new Promise((r) => setTimeout(r, 2000));
        completed++;
        completionTimes.push(Date.now());
      }
    });

    const t0 = Date.now();
    const ids: (string | null)[] = [];
    // Enqueue 5 sequentially with the same singletonKey + DIFFERENT payloads
    // (deliberately exercises whether payload diff bypasses dedup).
    for (let i = 0; i < 5; i++) {
      const id = await boss.send(
        test_queue,
        { user_id: 'user-X', i },                      // differing payload per enqueue
        { singletonKey: 'user-X', singletonHours: 0 },
      );
      ids.push(id);
    }

    // Wait 6 seconds — enough for one slow job (2s) + a small queue tail.
    await new Promise((r) => setTimeout(r, 6500));

    // Forensic visibility — emit actual counts so a test failure documents the semantics.
    console.log(`[pgboss-singleton-1] enqueued=5, ids_returned_non_null=${ids.filter(Boolean).length}, completed=${completed}, time_window=${Date.now() - t0}ms`);

    // Documented intent: at most 1 in-flight + at most 1 backlog ⇒ ≤2 completions in 6s.
    // If pg-boss returns null for some send() calls (already-deduped at enqueue), even better:
    // ids.filter(Boolean).length should be ≤ 2.
    expect(
      completed,
      `Expected ≤2 completions per pg-boss documented dedup intent; got ${completed}. ` +
      `If this fires consistently, payload-diff bypasses singleton dedup in pg-boss v12 — ` +
      `add an advisory DB lock at sync entry: SELECT pg_try_advisory_xact_lock(hashtext('pluggy.sync.' || user_id)).`,
    ).toBeLessThanOrEqual(2);
  }, 15000);

  it('pgboss-singleton-2: different singletonKeys run concurrently', async () => {
    const boss = await getBoss();
    let completed = 0;
    const concurrent_witnesses: number[] = [];

    const test_queue = `pluggy.sync.testB.${Date.now()}`;
    await boss.createQueue(test_queue);
    let in_flight = 0;
    let max_concurrent = 0;
    await boss.work(test_queue, { localConcurrency: 4 }, async (jobs: Job<unknown>[]) => {
      for (const _ of jobs) {
        in_flight++;
        max_concurrent = Math.max(max_concurrent, in_flight);
        concurrent_witnesses.push(in_flight);
        await new Promise((r) => setTimeout(r, 1000));
        in_flight--;
        completed++;
      }
    });

    // 3 jobs for user-A and 3 for user-B (singleton dedup is per-key).
    for (let i = 0; i < 3; i++) {
      await boss.send(test_queue, { user_id: 'user-A', i }, { singletonKey: 'user-A', singletonHours: 0 });
      await boss.send(test_queue, { user_id: 'user-B', i }, { singletonKey: 'user-B', singletonHours: 0 });
    }

    await new Promise((r) => setTimeout(r, 5000));

    console.log(`[pgboss-singleton-2] completed=${completed}, max_concurrent=${max_concurrent}, witnesses=${concurrent_witnesses.join(',')}`);

    expect(
      max_concurrent,
      `Expected at least 2 concurrent jobs (one per user-key); got ${max_concurrent}. ` +
      `If max_concurrent=1 the dedup is global, not per-key — singletonKey is misconfigured.`,
    ).toBeGreaterThanOrEqual(2);
  }, 10000);
});

import { describe, it, expect, vi, beforeEach } from 'vitest';

describe('vi.doMock + dynamic import stability under singleton testcontainer (Concern #10)', () => {
  beforeEach(() => {
    vi.resetModules();
    vi.unmock('@/services/PluggyService');     // clear any prior mock from sibling suites
  });

  it('mock-stability-1: 3 sequential cases each see their fresh mock', async () => {
    // Case A
    const fetchA = vi.fn().mockResolvedValue({ results: [{ id: 'A', name: 'A' }] });
    vi.doMock('@/services/PluggyService', () => ({
      getPluggyService: () => ({ fetchAccounts: fetchA }),
    }));
    const modA = await import('@/services/PluggyService');
    const resA = await modA.getPluggyService().fetchAccounts({ /* ... */ } as never);
    expect(resA.results[0].id).toBe('A');
    expect(fetchA).toHaveBeenCalledTimes(1);

    // Case B — must reset + redo
    vi.resetModules();
    vi.doMock('@/services/PluggyService', () => ({
      getPluggyService: () => ({ fetchAccounts: vi.fn().mockResolvedValue({ results: [{ id: 'B', name: 'B' }] }) }),
    }));
    const modB = await import('@/services/PluggyService');
    const resB = await modB.getPluggyService().fetchAccounts({} as never);
    expect(resB.results[0].id).toBe('B');

    // Case C
    vi.resetModules();
    vi.doMock('@/services/PluggyService', () => ({
      getPluggyService: () => ({ fetchAccounts: vi.fn().mockResolvedValue({ results: [{ id: 'C', name: 'C' }] }) }),
    }));
    const modC = await import('@/services/PluggyService');
    const resC = await modC.getPluggyService().fetchAccounts({} as never);
    expect(resC.results[0].id).toBe('C');
  });

  it('mock-stability-2 (negative): omitting vi.resetModules between cases produces stale mock — documents the pitfall', async () => {
    const fetchA = vi.fn().mockResolvedValue({ results: [{ id: 'A' }] });
    vi.doMock('@/services/PluggyService', () => ({
      getPluggyService: () => ({ fetchAccounts: fetchA }),
    }));
    const modA = await import('@/services/PluggyService');
    await modA.getPluggyService().fetchAccounts({} as never);

    // INTENTIONALLY skip vi.resetModules — re-doMock without reset.
    vi.doMock('@/services/PluggyService', () => ({
      getPluggyService: () => ({ fetchAccounts: vi.fn().mockResolvedValue({ results: [{ id: 'B' }] }) }),
    }));
    const modB = await import('@/services/PluggyService');
    const resB = await modB.getPluggyService().fetchAccounts({} as never);

    // The stale-mock pitfall: modA and modB are the SAME cached module — second
    // doMock is ignored. resB should still show 'A', not 'B'.
    expect(resB.results[0].id, 'pitfall: stale module cache returns first mock').toBe('A');
  });

# Pluggy Test Conventions

**Status:** ACTIVE
**Closes:** 02-REVIEWS.md Concerns #8 (pg-boss singleton) + #10 (vi.doMock stability)

## Singleton Testcontainer (plan 02-09 baseline)

- `pool: 'forks'`, `singleFork: true`, `isolate: false` for the integration project.
- One Postgres container booted via globalSetup; shared across all suites.
- Module-level state (DB client, pg-boss handle) lives on `globalThis` to survive `vi.resetModules()`.

## Mocking PluggyService in Route Tests

**Canonical pattern (USE THIS):**

```typescript
beforeEach(() => {
  vi.resetModules();
  vi.unmock('@/services/PluggyService');
});

it('route under test', async () => {
  vi.doMock('@/services/PluggyService', () => ({
    getPluggyService: () => ({ fetchAccounts: vi.fn().mockResolvedValue({ ... }) }),
  }));
  const { POST } = await import('@/app/api/pluggy/items/route');     // dynamic import AFTER doMock
  // exercise POST(...)
});
```

**Why:**

1. `vi.doMock` is hoisted-by-mock vs `vi.mock` which is hoisted-by-statement. We use `doMock` so we can change the mock per-case.
2. Dynamic `await import(...)` AFTER `doMock` ensures the import sees the current mock — a top-level static import would have been resolved before the doMock call.
3. `vi.resetModules()` in `beforeEach` flushes the module cache so each case starts with no prior mock applied.
4. `vi.unmock(...)` clears any leftover mock registration from a sibling suite (under `isolate: false`, mock registrations can leak between files).

**Anti-pattern (DO NOT DO):**

```typescript
import { POST } from '@/app/api/pluggy/items/route';   // static import resolves with REAL service
beforeEach(() => {
  vi.doMock('@/services/PluggyService', () => ({ ... }));   // too late — POST already imported
});
```

The negative test in `tests/integration/pluggy/route-mock-stability.test.ts` (mock-stability-2) demonstrates the pitfall.

## pg-boss Singleton Semantics (Concern #8)

Plan 02-16's `tests/integration/jobs/pg-boss-singleton.test.ts` empirically verifies that `singletonKey + singletonHours=0` produces ≤2 completions per key when 5 jobs are enqueued. If a future pg-boss upgrade breaks this assumption, the test FAILS and the SUMMARY documents the new semantics. The fallback is a Postgres advisory lock at sync entry:

```sql

```

This is a follow-up plan, not implemented in 02-16.

The shared testcontainer means table data persists across suites. Each integration suite MUST `TRUNCATE` the tables it touches in `beforeEach` (or use a savepoint pattern). The current Phase 2 follow-up tracker (STATE.md) calls out a per-suite truncation gap; this conventions doc is the authoritative reference once that gap closes.
