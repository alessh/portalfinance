**Step 1 — `src/jobs/workers/transferDetectorWorker.ts`** (D-33 — deterministic 4-invariant SQL):
```typescript
import type { Job } from 'pg-boss';
import { sql } from 'drizzle-orm';
import { db } from '@/db';
import { transactions } from '@/db/schema';
import { logger } from '@/lib/logger';
import { hashUserIdForSentry as hashId } from '@/lib/sentry';
import { recordAudit } from '@/lib/auditLog';

interface Payload { user_id: string; }

export async function transferDetectorWorker(jobs: Job<Payload>[]): Promise<void> {
  for (const job of jobs) {
    const user_id = job.data.user_id;
    try {
      // D-33: same |amount|, opposite type, same user, two different account_id, ≤3 days posted_at delta
      // Run as a single SQL statement that pairs candidates and updates both legs atomically.
      // Returns the count of newly-flagged pairs.
      const result = await db.execute(sql`
        WITH candidates AS (
          SELECT t1.id AS id_a, t2.id AS id_b
          FROM ${transactions} t1
          INNER JOIN ${transactions} t2
            ON t1.user_id = t2.user_id
            AND t2.user_id = ${user_id}
            AND t1.id < t2.id
            AND t1.is_transfer = false
            AND t2.is_transfer = false
            AND t1.account_id <> t2.account_id
            AND t1.amount = t2.amount
            AND t1.type <> t2.type
            AND ABS(EXTRACT(EPOCH FROM (t1.posted_at - t2.posted_at))) <= 3 * 24 * 60 * 60
        ),
        updated AS (
          UPDATE ${transactions} t
          SET is_transfer = true,
              transfer_pair_id = CASE
                WHEN t.id = c.id_a THEN c.id_b
                WHEN t.id = c.id_b THEN c.id_a
              END,
              updated_at = now()
          FROM candidates c
          WHERE t.id IN (c.id_a, c.id_b)
          RETURNING t.id
        )
        SELECT count(*)::int AS flagged FROM updated
      `);
      const flagged = (result as { flagged: number }[] | { rows: { flagged: number }[] })?.[0]?.flagged
        ?? ((result as { rows: { flagged: number }[] }).rows?.[0]?.flagged ?? 0);

      logger.info({
        event: 'transfer_detected',
        user_id_hashed: hashId(user_id),
        count: flagged,
      }, 'transfer detector run');

      if (flagged > 0) {
        await recordAudit({
          user_id,
          action: 'transfer_detected',
          metadata: { count: flagged },
        });
      }
    } catch (err) {
      logger.error({ event: 'worker_job_failed', job_id: job.id, worker: 'transferDetector', error: String(err) }, 'failed');
      throw err;
    }
  }
}
```
**Note on Drizzle `db.execute()` return shape:** depending on the driver (postgres-js vs node-postgres), the result may be the array directly OR `{ rows: [...] }`. The unwrap above tolerates both — confirm during implementation and tighten if needed; the Drizzle docs distinguish per-driver shape.

**Step 2 -- `src/jobs/workers/faturaDetectorWorker.ts`** (P8 / TX-05 -- checking debit ~= credit-card balance near due date, with the +/-7-day proximity window required by Pitfall P8 / TX-05):
```typescript
import type { Job } from 'pg-boss';
import { and, eq, gte, lte, sql } from 'drizzle-orm';
import { db } from '@/db';
import { accounts, transactions } from '@/db/schema';
import { logger } from '@/lib/logger';
import { hashUserIdForSentry as hashId } from '@/lib/sentry';
import { recordAudit } from '@/lib/auditLog';

interface Payload { user_id: string; }

export async function faturaDetectorWorker(jobs: Job<Payload>[]): Promise<void> {
  for (const job of jobs) {
    const user_id = job.data.user_id;
    try {
      // P8 / TX-05 -- a checking-account DEBIT t1 is flagged is_credit_card_payment=true iff:
      //   (a) |t1.amount| equals the credit-card account's balance at sync time, AND
      //   (b) t1.posted_at is within +/-7 days of the credit-card account's accounts.updated_at
      //       (proximity proxy for the billing-cycle close), AND
      //   (c) the credit-card account is the SAME user_id (already enforced by user scope).
      // Fallback (commented inline below): if accounts.updated_at proves unreliable, widen to a
      // +/-10-day window AND require equality with a DEBIT-aggregate sum on the credit-card account
      // in the same window. Phase 6 follow-up will replace this with Pluggy's
      // creditData.balanceDueDate / balanceCloseDate when reliably populated.
      const result = await db.execute(sql`
        WITH cc_balances AS (
          SELECT a.id AS cc_account_id, a.user_id, a.balance, a.credit_limit, a.updated_at AS cc_updated_at
          FROM ${accounts} a
          WHERE a.user_id = ${user_id} AND a.type = 'CREDIT_CARD' AND a.status = 'ACTIVE'
        ),
        candidates AS (
          SELECT t.id AS tx_id, t.amount AS tx_amount, cc.cc_account_id
          FROM ${transactions} t
          INNER JOIN ${accounts} chk ON chk.id = t.account_id AND chk.type IN ('CHECKING','SAVINGS')
          INNER JOIN cc_balances cc
            ON cc.user_id = t.user_id
            AND cc.balance = t.amount
            -- (b) +/-7-day proximity using accounts.updated_at as the billing-cycle proxy.
            -- Fallback (10 * 24 * 60 * 60 = 10-day widen + DEBIT-aggregate sum equality) is left as
            -- this comment block so future maintainers can enable it without re-litigating the heuristic.
            AND ABS(EXTRACT(EPOCH FROM (t.posted_at - cc.cc_updated_at))) <= 7 * 24 * 60 * 60
          WHERE t.user_id = ${user_id}
            AND t.type = 'DEBIT'
            AND t.is_credit_card_payment = false
            AND t.is_transfer = false
        ),
        updated AS (
          UPDATE ${transactions} t
          SET is_credit_card_payment = true, updated_at = now()
          FROM candidates c
          WHERE t.id = c.tx_id
          RETURNING t.id
        )
        SELECT count(*)::int AS flagged FROM updated
      `);
      const flagged = (result as { rows?: { flagged: number }[] })?.rows?.[0]?.flagged
        ?? (result as { flagged: number }[])?.[0]?.flagged ?? 0;

      logger.info({
        event: 'fatura_detected',
        user_id_hashed: hashId(user_id),
        count: flagged,
      }, 'fatura detector run');

      if (flagged > 0) {
        await recordAudit({
          user_id,
          action: 'fatura_detected',
          metadata: { count: flagged },
        });
      }
    } catch (err) {
      logger.error({ event: 'worker_job_failed', job_id: job.id, worker: 'faturaDetector', error: String(err) }, 'failed');
      throw err;
    }
  }
}
```
**Note:** Phase 2's fatura heuristic uses the +/-7-day proximity window per Pitfall P8 / TX-05 (per checker dispositions). The fallback +/-10-day wider window using DEBIT-aggregate sum equality is left as an inline code-comment fallback for the accounts.updated_at-unavailable case.

**Phase 6 follow-up (record verbatim in 02-05-SUMMARY.md):** "TX-05 fatura precision -- extend with Pluggy `creditData.balanceDueDate` (and optionally `balanceCloseDate`) when those fields become reliably populated; tighten the proximity window from +/-7 days to +/-3 days around the actual due date." This is a deliberate, conservative Phase 2 deferral, not a scope reduction.

**Step 4 — `src/jobs/workers/reAuthNotifierWorker.ts`** (D-34 24h debounce + D-35 send):
```typescript
import type { Job } from 'pg-boss';
import { eq } from 'drizzle-orm';
import { db } from '@/db';
import { pluggy_items, users } from '@/db/schema';
import { sendEmail } from '@/lib/mailer';
import { logger } from '@/lib/logger';
import { hashUserIdForSentry as hashId } from '@/lib/sentry';
import { env } from '@/lib/env';
import { ReAuthRequired, renderReAuthRequiredText } from '@/emails/ReAuthRequired';
import * as React from 'react';
import { recordAudit } from '@/lib/auditLog';

const TWENTY_FOUR_HOURS_MS = 24 * 60 * 60 * 1000;

interface Payload { webhook_event_id?: string; item_id_pluggy?: string; item_id?: string; }

export async function reAuthNotifierWorker(jobs: Job<Payload>[]): Promise<void> {
  for (const job of jobs) {
    try {
      // Resolve internal pluggy_items row (mirror plan 02-04 worker logic)
      let item: typeof pluggy_items.$inferSelect | undefined;
      if (job.data.item_id) {
        item = await db.query.pluggy_items.findFirst({ where: eq(pluggy_items.id, job.data.item_id) });
      } else if (job.data.item_id_pluggy) {
        const { hashPluggyItemId } = await import('@/lib/crypto');   // RESEARCH OQ#6 RESOLVED -- HMAC, NOT bare SHA-256
        const hash = hashPluggyItemId(job.data.item_id_pluggy);
        item = await db.query.pluggy_items.findFirst({ where: eq(pluggy_items.pluggy_item_id_hash, hash) });
      }
      if (!item) return;

      // D-34 debounce
      if (item.last_reauth_email_at && item.last_reauth_email_at.getTime() > Date.now() - TWENTY_FOUR_HOURS_MS) {
        logger.info({
          event: 'reconnect_email_debounced',
          item_id_hashed: hashId(item.id),
          debounce_seconds: Math.floor((Date.now() - item.last_reauth_email_at.getTime()) / 1000),
        }, 'reconnect email debounced — within 24h window');
        return;
      }

      const user = await db.query.users.findFirst({ where: eq(users.id, item.user_id), columns: { email: true } });
      if (!user) return;

      const reconnect_url = `${env.NEXTAUTH_URL}/connect?reconnect=${item.id}`;
      const props = {
        institution_name: item.institution_name,
        last_synced_at: item.last_synced_at ?? item.created_at,
        reconnect_url,
      };
      await sendEmail({
        to: user.email,
        subject: `Reconecte seu ${item.institution_name}`,
        template: React.createElement(ReAuthRequired, props),
        plaintext: renderReAuthRequiredText(props),
      });

      await db.update(pluggy_items).set({ last_reauth_email_at: new Date(), updated_at: new Date() }).where(eq(pluggy_items.id, item.id));

      logger.info({
        event: 'reconnect_email_sent',
        item_id_hashed: hashId(item.id),
        debounce_seconds: 0,
      }, 'reconnect email sent');

      await recordAudit({
        user_id: item.user_id,
        action: 'item_reauth_started',
        metadata: { connector_id: item.connector_id, institution_name: item.institution_name, cooldown_bypassed: false },
      });
    } catch (err) {
      logger.error({ event: 'worker_job_failed', job_id: job.id, worker: 'reAuthNotifier', error: String(err) }, 'failed');
      throw err;
    }
  }
}
```

**Step 5 — `src/jobs/workers/reconcileStaleItemsWorker.ts`** (D-38 hourly cron):
```typescript
import type { Job } from 'pg-boss';
import { sql } from 'drizzle-orm';
import { db } from '@/db';
import { pluggy_items } from '@/db/schema';
import { enqueue, QUEUES } from '@/jobs/boss';
import { logger } from '@/lib/logger';

export async function reconcileStaleItemsWorker(jobs: Job<unknown>[]): Promise<void> {
  for (const job of jobs) {
    try {
      // D-38: items where last_synced_at < now() - 12h AND status NOT IN ('LOGIN_ERROR', 'WAITING_USER_INPUT')
      const stale = await db.execute<{ id: string; user_id: string }>(sql`
        SELECT id, user_id FROM ${pluggy_items}
        WHERE (last_synced_at IS NULL OR last_synced_at < now() - interval '12 hours')
          AND status NOT IN ('LOGIN_ERROR', 'WAITING_USER_INPUT')
      `);
      const rows = (stale as { rows?: Array<{ id: string; user_id: string }> }).rows ?? (stale as Array<{ id: string; user_id: string }>);
      for (const row of rows) {
        await enqueue(
          QUEUES.PLUGGY_SYNC,
          { user_id: row.user_id, item_id: row.id, trigger: 'reconcile' },
          { singletonKey: row.user_id, singletonHours: 0 },
        );
      }
      if (rows.length > 5) {
        logger.warn({ event: 'pluggy_reconcile_high_stale_count', count: rows.length }, 'reconciliation queued >5 stale items — investigate');
      } else {
        logger.info({ event: 'pluggy_reconcile_run', count: rows.length }, 'reconciliation cron tick');
      }
    } catch (err) {
      logger.error({ event: 'worker_job_failed', job_id: job.id, worker: 'reconcileStaleItems', error: String(err) }, 'failed');
      throw err;
    }
  }
}
```
