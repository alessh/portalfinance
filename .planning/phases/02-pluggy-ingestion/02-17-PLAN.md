**Step 2 — Create `tests/e2e/pluggy/sandbox-connect.spec.ts`:**

```typescript
import { test, expect } from '@playwright/test';

// Concern #2 — real Pluggy sandbox gate for roadmap success criterion #1.
// This spec runs in the nightly GitHub workflow with PLUGGY_SANDBOX_CLIENT_ID/SECRET
// populated from repo secrets. It is SKIPPED locally unless those env vars are set
// (avoid burning Pluggy sandbox quota on every developer push).

test.describe('Pluggy sandbox real connect (success criterion #1 hard gate)', () => {
  test.skip(
    !process.env.PLUGGY_SANDBOX_CLIENT_ID || !process.env.PLUGGY_SANDBOX_CLIENT_SECRET,
    'PLUGGY_SANDBOX_CLIENT_ID/SECRET not set — skipping real-sandbox gate; runs nightly in CI',
  );

  test('connect → /transactions shows >=1 transaction within 60 seconds (real sandbox)', async ({ page }) => {
    // 1. Sign up + sign in (inline canonical pattern — see <interfaces> note above).
    //    Sandbox CPF: 761.092.776-73 (D-48 — Pluggy sandbox basic OF flow).
    const test_email = `e2e+sandbox+${Date.now()}@portalfinance.app`;
    const test_password = 'Correct-Horse-1234';
    await page.goto('/signup');
    await page.fill('input[name="email"]', test_email);
    await page.fill('input[name="password"]', test_password);
    await page.fill('input[name="confirmPassword"]', test_password);
    await page.getByText('Li e concordo').click();
    await page.click('button[type="submit"]');
    await page.waitForURL('**/dashboard', { timeout: 30_000 });

    // 2. Navigate to /connect.
    await page.goto('/connect');

    // 3. Fill CPF on the consent screen.
    await page.fill('input[id=cpf]', '761.092.776-73');

    // 4. Check consent.
    await page.check('input[type=checkbox]');

    // 5. Submit — opens Pluggy widget iframe.
    await page.click('button:has-text("Concordar")');

    // 6. Inside Pluggy widget iframe: pick the sandbox bank, enter user-ok / password-ok.
    //    The widget runs in its own iframe; we use page.frameLocator.
    const widget_frame = page.frameLocator('iframe[title*="Pluggy" i], iframe[src*="pluggy"]');
    await widget_frame.locator('text=Pluggy Bank').first().click({ timeout: 30000 });
    await widget_frame.locator('input[name=user]').fill('user-ok');
    await widget_frame.locator('input[type=password]').fill('password-ok');
    await widget_frame.locator('button:has-text("Continue"), button:has-text("Continuar")').first().click();

    // 7. After widget onSuccess → redirect to /connect/success → polling → /transactions.
    //    Start the 60s timer at this point.
    const t_widget_done = Date.now();
    await expect(page).toHaveURL(/\/transactions/, { timeout: 65000 });

    // 8. Assert at least 1 transaction visible.
    //    Selector mirrors UI-SPEC § 3.5 (TransactionList rows).
    const tx_rows = page.locator('[data-testid=transaction-row], li[data-tx-id]');
    await expect(tx_rows.first()).toBeVisible({ timeout: 60000 - (Date.now() - t_widget_done) });

    const observed_latency_ms = Date.now() - t_widget_done;
    const tx_count = await tx_rows.count();

    // Forensic — surface to workflow logs for the audit-trail file.
    console.log(`[sandbox-connect] observed_latency_ms=${observed_latency_ms} tx_count=${tx_count}`);

    expect(observed_latency_ms, `connect → first transaction visible took ${observed_latency_ms}ms; criterion #1 = 60000ms`).toBeLessThan(60000);
    expect(tx_count).toBeGreaterThanOrEqual(1);
  });
});
```
