**Step 3 — `src/components/transactions/TransactionList.tsx`** (UI-SPEC § 3.5 verbatim):
```typescript
'use client';
import { format, isToday, isYesterday } from 'date-fns';
import { ptBR } from 'date-fns/locale';
import { Button } from '@/components/ui/button';
import { Loader2 } from 'lucide-react';

export interface TransactionRow {
  id: string;
  description: string;
  amount: string;          // Decimal as string from Drizzle numeric
  type: 'DEBIT' | 'CREDIT';
  posted_at: Date;
  status: 'PENDING' | 'POSTED';
  is_transfer: boolean;
  is_credit_card_payment: boolean;
  account_name: string;
}

const fmtBRL = new Intl.NumberFormat('pt-BR', { style: 'currency', currency: 'BRL' });

function dateGroupLabel(d: Date): string {
  if (isToday(d)) return 'Hoje';
  if (isYesterday(d)) return 'Ontem';
  return format(d, 'd MMM', { locale: ptBR });
}

function hashColor(name: string): string {
  const colors = ['bg-slate-400', 'bg-violet-400', 'bg-orange-400', 'bg-sky-400', 'bg-pink-400', 'bg-lime-400'];
  let h = 0; for (const c of name) h = (h * 31 + c.charCodeAt(0)) | 0;
  return colors[Math.abs(h) % colors.length];
}

export function TransactionList({ transactions, hasMore, onLoadMore, isLoadingMore }: {
  transactions: TransactionRow[];
  hasMore: boolean;
  onLoadMore: () => void;
  isLoadingMore: boolean;
}) {
  // Group by day (ISO date)
  const groups = new Map<string, TransactionRow[]>();
  for (const t of transactions) {
    const key = format(t.posted_at, 'yyyy-MM-dd');
    if (!groups.has(key)) groups.set(key, []);
    groups.get(key)!.push(t);
  }
  return (
    <ul className="space-y-0">
      {[...groups.entries()].map(([key, rows]) => (
        <li key={key}>
          <h2 className="sticky h-8 px-4 bg-muted text-xs font-semibold text-muted-foreground uppercase tracking-wide flex items-center"
              style={{ top: 'var(--sticky-offset, 56px)' }}>
            {dateGroupLabel(rows[0].posted_at)}
          </h2>
          <ul>
            {rows.map((t) => {
              const sign = t.type === 'CREDIT' ? '+' : '-';
              const amount_label = `${sign}${fmtBRL.format(Number(t.amount))}`;
              const amt_color = t.type === 'CREDIT' ? 'text-emerald-700 dark:text-emerald-300' : 'text-foreground';
              const dim = t.status === 'PENDING' ? 'opacity-60' : '';
              return (
                <li key={t.id} className={`flex items-start gap-2 min-h-[56px] px-4 py-2 ${dim}`}>
                  <span className={`h-2 w-2 rounded-full ${hashColor(t.account_name)} mt-2`} aria-hidden="true" />
                  <div className="flex-1 min-w-0">
                    <p className="text-sm text-foreground truncate max-w-[60%]">{t.description}</p>
                    <p className="text-xs text-muted-foreground flex items-center gap-2">
                      {t.account_name}
                      {t.status === 'PENDING' && <span className="text-xs px-2 py-1 rounded-sm bg-amber-100 text-amber-700 dark:bg-amber-950 dark:text-amber-300 font-medium" aria-label="Pendente">Pendente</span>}
                      {t.is_transfer && <span className="text-xs px-2 py-1 rounded-sm bg-muted text-muted-foreground font-medium" aria-label="Transferência">Transferência</span>}
                      {t.is_credit_card_payment && <span className="text-xs px-2 py-1 rounded-sm bg-muted text-muted-foreground font-medium" aria-label="Pagamento de fatura">Pagamento de fatura</span>}
                    </p>
                  </div>
                  <span className={`text-sm font-semibold tabular-nums ${amt_color}`} aria-label={`Valor: ${amount_label} ${t.type === 'CREDIT' ? 'crédito' : 'débito'}`}>
                    {amount_label}
                  </span>
                </li>
              );
            })}
          </ul>
        </li>
      ))}
      {hasMore && (
        <li className="text-center py-6">
          <Button variant="outline" onClick={onLoadMore} disabled={isLoadingMore} className="min-w-[200px]">
            {isLoadingMore ? <><Loader2 className="animate-spin h-4 w-4 mr-2" /> Carregando...</> : 'Carregar mais'}
          </Button>
        </li>
      )}
    </ul>
  );
}
```

**Step 1 — `src/app/api/pluggy/items/[id]/sync/route.ts`** (D-26, D-28, D-29, D-30, CONN-06):
```typescript
export const runtime = 'nodejs';

import { NextResponse } from 'next/server';
import { and, eq } from 'drizzle-orm';
import { db } from '@/db';
import { pluggy_items, users } from '@/db/schema';
import { requireSession } from '@/lib/session';
import { enqueue, QUEUES } from '@/jobs/boss';
import { recordAudit } from '@/lib/auditLog';

const COOLDOWN_MS = 30 * 60 * 1000;

export async function POST(
  req: Request,
  { params }: { params: Promise<{ id: string }> },
): Promise<Response> {
  const { id } = await params;
  const session = await requireSession(req);

  // Load item + tier in one read (IDOR via user_id filter)
  const row = await db
    .select({
      item_id: pluggy_items.id,
      last_synced_at: pluggy_items.last_synced_at,
      tier: users.subscription_tier,
      status: pluggy_items.status,
    })
    .from(pluggy_items)
    .innerJoin(users, eq(users.id, pluggy_items.user_id))
    .where(and(eq(pluggy_items.id, id), eq(pluggy_items.user_id, session.userId)))
    .limit(1);

  if (row.length === 0) return NextResponse.json({ error: 'NOT_FOUND' }, { status: 404 });    // P26
  const it = row[0];

  // D-29: free-tier hard block
  if (it.tier === 'free') {
    return NextResponse.json({ error: 'PAYWALL', upgrade_url: '/settings/billing' }, { status: 403 });
  }

  // D-28: cooldown
  const last = it.last_synced_at?.getTime() ?? 0;
  const elapsed_ms = Date.now() - last;
  if (last > 0 && elapsed_ms < COOLDOWN_MS) {
    const retry_after_seconds = Math.ceil((COOLDOWN_MS - elapsed_ms) / 1000);
    return NextResponse.json(
      { error: 'COOLDOWN_ACTIVE', retry_after_seconds },
      { status: 429, headers: { 'Retry-After': String(retry_after_seconds) } },
    );
  }

  // Enqueue with per-user singletonKey
  await enqueue(
    QUEUES.PLUGGY_SYNC,
    { user_id: session.userId, item_id: it.item_id, trigger: 'manual' },
    { singletonKey: session.userId, singletonHours: 0 },
  );

  await recordAudit({
    user_id: session.userId,
    action: 'manual_sync_triggered',
    metadata: { item_id: it.item_id, cooldown_bypassed: false },
    ip_address: req.headers.get('x-forwarded-for'),
    user_agent: req.headers.get('user-agent'),
  });

  return NextResponse.json({ accepted: true }, { status: 202 });
}
```

**Step 2 — `src/app/api/pluggy/items/[id]/route.ts`** (DELETE — D-04, LGPD-02):
```typescript
export const runtime = 'nodejs';

import { NextResponse } from 'next/server';
import { and, eq } from 'drizzle-orm';
import { db } from '@/db';
import { accounts, pluggy_items, user_consents } from '@/db/schema';
import { requireSession } from '@/lib/session';
import { getPluggyService } from '@/services/PluggyService';
import { recordAudit } from '@/lib/auditLog';
import { getConsentVersionHash } from '@/lib/consentVersions';

export async function DELETE(
  req: Request,
  { params }: { params: Promise<{ id: string }> },
): Promise<Response> {
  const { id } = await params;
  const session = await requireSession(req);

  const it = await db.query.pluggy_items.findFirst({
    where: (pi, { and, eq }) => and(eq(pi.id, id), eq(pi.user_id, session.userId)),
    columns: { id: true, pluggy_item_id_enc: true, connector_id: true, institution_name: true },
  });
  if (!it) return NextResponse.json({ error: 'NOT_FOUND' }, { status: 404 });

  // 1. Call Pluggy DELETE first — if this fails, do NOT mutate local state (D-04 atomicity)
  try {
    await getPluggyService().deleteItem({ user_id: session.userId, item_id_enc: it.pluggy_item_id_enc });
  } catch (err) {
    return NextResponse.json({ error: 'PLUGGY_API_ERROR', message: String(err) }, { status: 502 });
  }

  // 2. Soft-delete: mark all accounts under this item as DELETED (cascade by user choice; transactions remain readable per D-04)
  await db.update(accounts).set({ status: 'DELETED', updated_at: new Date() }).where(eq(accounts.pluggy_item_id, it.id));

  // 3. Append-only consent revocation (LGPD-02 + D-04)
  await db.insert(user_consents).values({
    user_id: session.userId,
    scope: `PLUGGY_CONNECTOR:${it.connector_id}`,
    action: 'REVOKED',
    consent_version_hash: getConsentVersionHash('pluggy_connect_v1'),
    ip_address: req.headers.get('x-forwarded-for'),
    user_agent: req.headers.get('user-agent'),
  });

  // 4. Audit
  await recordAudit({
    user_id: session.userId,
    action: 'item_disconnected',
    metadata: { connector_id: it.connector_id, institution_name: it.institution_name },
    ip_address: req.headers.get('x-forwarded-for'),
    user_agent: req.headers.get('user-agent'),
  });

  return NextResponse.json({ disconnected: true });
}
```
