**Step 1 — `src/app/api/webhooks/pluggy/route.ts`** (D-42, PATTERNS.md verbatim shape, Pitfall 10 unknown-event handling):
```typescript
export const runtime = 'nodejs';

import { timingSafeEqual } from 'node:crypto';
import { db } from '@/db';
import { webhook_events } from '@/db/schema';
import { enqueue, QUEUES } from '@/jobs/boss';
import { env } from '@/lib/env';
import { logger } from '@/lib/logger';

interface PluggyEvent {
  event: string;
  eventId: string;
  itemId?: string;
  [k: string]: unknown;
}

function mapEventToQueue(event_type: string): string | null {
  switch (event_type) {
    case 'item/created':
    case 'item/updated':
    case 'item/login_succeeded':
    case 'transactions/created':
    case 'transactions/updated':
    case 'transactions/deleted':
      return QUEUES.PLUGGY_SYNC;
    case 'item/error':
    case 'item/waiting_user_input':
      return QUEUES.PLUGGY_REAUTH_NOTIFIER;
    case 'item/deleted':
    case 'connector/status_updated':
      return null;        // explicit no-op
    default:
      return null;        // Pitfall 10 — store row, do not enqueue
  }
}

export async function POST(req: Request): Promise<Response> {
  const start = Date.now();

  // 1. Verify custom signature header (D-42, Pitfall P3)
  const sig_header = req.headers.get('x-pluggy-signature') ?? '';
  const expected = env.PLUGGY_WEBHOOK_SECRET ?? '';
  const sig_buf = Buffer.from(sig_header);
  const exp_buf = Buffer.from(expected);
  if (sig_buf.length !== exp_buf.length || !timingSafeEqual(sig_buf, exp_buf)) {
    logger.warn({ event: 'pluggy_webhook_signature_failed' }, 'invalid signature');
    return new Response('unauthorized', { status: 401 });
  }

  // 2. Parse JSON body
  let body: PluggyEvent;
  try {
    body = (await req.json()) as PluggyEvent;
  } catch {
    return new Response('bad json', { status: 400 });
  }
  if (!body.event || !body.eventId) {
    return new Response('bad payload', { status: 400 });
  }

  // 3. Idempotent INSERT
  const inserted = await db
    .insert(webhook_events)
    .values({
      source: 'PLUGGY',
      event_type: body.event,
      event_id: body.eventId,
      payload: body as Record<string, unknown>,
    })
    .onConflictDoNothing()
    .returning({ id: webhook_events.id });

  const was_duplicate = inserted.length === 0;

  // 4. Enqueue if mapped + new
  if (!was_duplicate) {
    const queue = mapEventToQueue(body.event);
    if (queue) {
      // D-30: reconnect path -- when Pluggy fires item/login_succeeded, the user has just completed
      // the update-mode flow. Force trigger='reconnect' so the worker bypasses cooldown AND uses a
      // 12-month sync window (full re-fetch). Other events use the default trigger='webhook'.
      const trigger = body.event === 'item/login_succeeded' ? 'reconnect' : 'webhook';
      await enqueue(queue, { webhook_event_id: inserted[0].id, item_id_pluggy: body.itemId, trigger });

      // D-13 + D-30: emit item_reauth_succeeded audit inline on item/login_succeeded BEFORE returning 200.
      // Resolve the internal user_id via pluggy_item_id_hash lookup (workers do the same; we duplicate here
      // because the audit MUST be timestamped at webhook receipt, not at worker dequeue).
      if (body.event === 'item/login_succeeded' && body.itemId) {
        const { recordAudit } = await import('@/lib/auditLog');
        const { hashPluggyItemId } = await import('@/lib/crypto');
        const { pluggy_items } = await import('@/db/schema');
        const { eq } = await import('drizzle-orm');
        const item_hash = hashPluggyItemId(body.itemId);
        const item = await db.query.pluggy_items.findFirst({
          where: eq(pluggy_items.pluggy_item_id_hash, item_hash),
          columns: { id: true, user_id: true },
        });
        if (item) {
          await recordAudit({
            user_id: item.user_id,
            action: 'item_reauth_succeeded',
            metadata: { item_id_hashed: item_hash.toString('hex') },   // hashed only, never plaintext (P4)
          });
        }
      }
    } else {
      logger.info({ event: 'pluggy_webhook_unmapped_event', event_type: body.event }, 'unmapped event stored only');
    }
  }

  // 5. Tail log + 200 response (must be <200ms — pg-boss send is fast)
  logger.info(
    {
      event: 'pluggy_webhook_received',
      event_type: body.event,
      event_id: body.eventId,
      latency_ms: Date.now() - start,
      was_duplicate,
    },
    'pluggy webhook processed',
  );

  return new Response('ok', { status: 200 });
}
```
**Note:** the enqueue payload includes `webhook_event_id` (UUID of the just-inserted row) and `item_id_pluggy` (the Pluggy plaintext item id from the webhook payload). The worker (Task 2) will look up the internal `pluggy_items.id` by hashing this Pluggy id — workers MUST never trust the webhook payload as authoritative; they re-fetch via PluggyService.

**Step 1 — `src/jobs/workers/pluggySyncWorker.ts`** (PATTERNS.md verbatim shape; cursor pagination per Specifics):
```typescript
import * as Sentry from '@sentry/nextjs';
import { sql } from 'drizzle-orm';
import type { Job } from 'pg-boss';
import { db } from '@/db';
import { accounts, pluggy_items, transactions } from '@/db/schema';
import { enqueue, QUEUES } from '@/jobs/boss';
import { logger } from '@/lib/logger';
import { hashUserIdForSentry as hashId } from '@/lib/sentry';
import { getPluggyService } from '@/services/PluggyService';
import { eq } from 'drizzle-orm';

interface SyncJobPayload {
  user_id?: string;                  // direct enqueue (from /api/pluggy/items)
  item_id?: string;                  // internal pluggy_items UUID
  webhook_event_id?: string;         // webhook-driven enqueue
  item_id_pluggy?: string;           // raw Pluggy id from webhook (untrusted)
  trigger?: 'first_connect' | 'webhook' | 'manual' | 'reconcile' | 'reconnect';
}

const TWELVE_MONTHS_MS = 365 * 24 * 60 * 60 * 1000;
const SEVEN_DAYS_MS = 7 * 24 * 60 * 60 * 1000;

export async function pluggySyncWorker(jobs: Job<SyncJobPayload>[]): Promise<void> {
  for (const job of jobs) {
    const start = Date.now();
    let item_row: typeof pluggy_items.$inferSelect | undefined;
    try {
      // 1. Resolve internal pluggy_items row — webhook path uses raw Pluggy id; direct path uses internal UUID.
      if (job.data.item_id) {
        item_row = await db.query.pluggy_items.findFirst({ where: eq(pluggy_items.id, job.data.item_id) });
      } else if (job.data.item_id_pluggy) {
        const { hashPluggyItemId } = await import('@/lib/crypto');   // RESEARCH OQ#6 RESOLVED -- HMAC, NOT bare SHA-256
        const hash = hashPluggyItemId(job.data.item_id_pluggy);
        item_row = await db.query.pluggy_items.findFirst({ where: eq(pluggy_items.pluggy_item_id_hash, hash) });
      }
      if (!item_row) {
        logger.warn({ event: 'sync_skipped', reason: 'item_not_found', job_id: job.id }, 'pluggy item not found');
        return;
      }

      // 2. Skip broken items (Pitfall P2)
      if (item_row.status === 'LOGIN_ERROR' || item_row.status === 'WAITING_USER_INPUT') {
        logger.warn({
          event: 'sync_failed',
          reason: 'item_broken',
          status: item_row.status,
          user_id_hashed: hashId(item_row.user_id),
          item_id_hashed: hashId(item_row.id),
        }, 'sync skipped — item not healthy');
        return;
      }

      const trigger = job.data.trigger ?? 'webhook';
      logger.info({
        event: 'sync_started',
        user_id_hashed: hashId(item_row.user_id),
        item_id_hashed: hashId(item_row.id),
        trigger,
      }, 'pluggy sync started');

      // D-30: reconnect path bypasses cooldown -- emit explicit audit row with cooldown_bypassed=true
      // BEFORE issuing any Pluggy SDK call. Auditors can correlate this row with the
      // 'item_reauth_succeeded' row emitted by the webhook receiver.
      if (trigger === 'reconnect') {
        const { recordAudit } = await import('@/lib/auditLog');
        await recordAudit({
          user_id: item_row.user_id,
          action: 'manual_sync_triggered',
          metadata: { item_id: item_row.id, cooldown_bypassed: true, trigger: 'reconnect' },
        });
      }

      // 3. Mark UPDATING (Sentry span wraps the rest of the work — D-47)
      await db.update(pluggy_items).set({ status: 'UPDATING', updated_at: new Date() }).where(eq(pluggy_items.id, item_row.id));

      const result = await Sentry.startSpan(
        { op: 'pluggy.sync', name: 'pluggy-sync-worker', attributes: { trigger } },
        async () => {
          const svc = getPluggyService();

          // 4. Fetch accounts → upsert
          const acct_resp = await svc.fetchAccounts({ user_id: item_row!.user_id, item_id_enc: item_row!.pluggy_item_id_enc });
          for (const a of acct_resp.results) {
            await db.insert(accounts).values({
              user_id: item_row!.user_id,
              pluggy_item_id: item_row!.id,
              pluggy_account_id: a.id,
              type: mapAccountType(a.type),
              subtype: a.subtype ?? null,
              name: a.name,
              currency: a.currencyCode ?? 'BRL',
              balance: String(a.balance),
              credit_limit: a.creditData?.creditLimit ? String(a.creditData.creditLimit) : null,
              owner: a.owner ?? null,
            }).onConflictDoUpdate({
              target: accounts.pluggy_account_id,
              set: {
                name: sql.raw('excluded.name'),
                balance: sql.raw('excluded.balance'),
                credit_limit: sql.raw('excluded.credit_limit'),
                updated_at: sql`now()`,
              },
            });
          }

          // 5. Sync window
          const now = new Date();
          const from_initial = new Date(now.getTime() - TWELVE_MONTHS_MS);
          const from_incremental = item_row!.last_synced_at
            ? new Date(item_row!.last_synced_at.getTime() - SEVEN_DAYS_MS)   // TX-02 overlap
            : from_initial;
          const from = trigger === 'first_connect' || trigger === 'reconnect' ? from_initial : from_incremental;

          // 6. Per-account cursor walk → upsert
          let added = 0, updated = 0;
          for (const a of acct_resp.results) {
            // Find local accounts.id for this pluggy_account_id
            const local_acct = await db.query.accounts.findFirst({ where: eq(accounts.pluggy_account_id, a.id), columns: { id: true } });
            if (!local_acct) continue;
            let cursor: string | undefined = undefined;
            do {
              const tx_resp = await svc.fetchTransactions({
                user_id: item_row!.user_id,
                item_id_enc: item_row!.pluggy_item_id_enc,
                account_id: a.id,
                from,
                to: now,
                cursor,
              });
              if (tx_resp.results.length > 0) {
                const rows = tx_resp.results.map((t) => ({
                  user_id: item_row!.user_id,
                  account_id: local_acct.id,
                  pluggy_transaction_id: t.id,
                  type: t.type,                              // 'DEBIT' | 'CREDIT' from Pluggy
                  amount: String(Math.abs(t.amount)),
                  currency: t.currencyCode ?? 'BRL',
                  description: t.description ?? '',
                  description_raw: t.descriptionRaw ?? null,
                  merchant_name: t.merchant?.name ?? null,
                  merchant_cnpj: t.merchant?.cnpj ?? null,
                  posted_at: new Date(t.date),
                  status: t.status === 'POSTED' ? 'POSTED' as const : 'PENDING' as const,
                  pluggy_category: t.category ?? null,
                  payment_method: t.paymentData?.paymentMethod ?? null,
                  raw_payload: t as Record<string, unknown>,
                }));
                const upserted = await db.insert(transactions).values(rows).onConflictDoUpdate({
                  target: transactions.pluggy_transaction_id,
                  set: {
                    status: sql.raw('excluded.status'),
                    amount: sql.raw('excluded.amount'),
                    description: sql.raw('excluded.description'),
                    description_raw: sql.raw('excluded.description_raw'),
                    posted_at: sql.raw('excluded.posted_at'),
                    raw_payload: sql.raw('excluded.raw_payload'),
                    updated_at: sql`now()`,
                    // DO NOT touch is_transfer / is_credit_card_payment / transfer_pair_id — set only by detectors
                  },
                }).returning({ id: transactions.id });
                added += upserted.length;     // overcounts updates as added in TX-01 sense; integration test asserts exact row count
              }
              cursor = tx_resp.next_cursor ?? undefined;
            } while (cursor);
          }

          // 7. Flip status, last_synced_at, clear errors
          await db.update(pluggy_items)
            .set({ status: 'UPDATED', last_synced_at: new Date(), updated_at: new Date() })
            .where(eq(pluggy_items.id, item_row!.id));

          // 8. Enqueue detectors
          await enqueue(QUEUES.PLUGGY_TRANSFER_DETECTOR, { user_id: item_row!.user_id });
          await enqueue(QUEUES.PLUGGY_FATURA_DETECTOR, { user_id: item_row!.user_id });

          return { transactions_added: added, transactions_updated: updated };
        },
      );

      logger.info({
        event: 'sync_completed',
        user_id_hashed: hashId(item_row.user_id),
        item_id_hashed: hashId(item_row.id),
        duration_ms: Date.now() - start,
        transactions_added: result.transactions_added,
        transactions_updated: result.transactions_updated,
      }, 'pluggy sync completed');
    } catch (err) {
      const status = (err as { sdk_status?: number; status?: number }).sdk_status ?? (err as { status?: number }).status;
      logger.error({
        event: 'sync_failed',
        reason: status === 429 ? 'rate_limited' : 'sdk_error',
        status,
        user_id_hashed: item_row ? hashId(item_row.user_id) : undefined,
        item_id_hashed: item_row ? hashId(item_row.id) : undefined,
        error: String(err),
      }, 'pluggy sync failed');
      if (item_row) {
        await db.update(pluggy_items).set({ last_error_at: new Date() }).where(eq(pluggy_items.id, item_row.id));
      }
      throw err;     // re-throw for pg-boss retry
    }
  }
}

function mapAccountType(t: string): 'CHECKING' | 'SAVINGS' | 'CREDIT_CARD' | 'LOAN' | 'INVESTMENT' | 'OTHER' {
  switch (t) {
    case 'BANK': return 'CHECKING';                  // Pluggy aggregates checking under BANK; subtype distinguishes
    case 'CREDIT': return 'CREDIT_CARD';
    case 'LOAN': return 'LOAN';
    case 'INVESTMENT': return 'INVESTMENT';
    default: return 'OTHER';
  }
}
```
