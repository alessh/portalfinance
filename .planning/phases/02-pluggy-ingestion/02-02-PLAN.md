**Step 0 — Extend `src/lib/crypto.ts` with `hashPluggyItemId()` helper (RESEARCH OQ#6 RESOLVED, mirrors `hashCPF()` shape):**
```typescript
import { createHmac } from 'node:crypto';
import { env } from '@/lib/env';

// Mirror of hashCPF() shape (Phase 1 plan 01-01) -- different pepper for defense-in-depth.
// Used everywhere a pluggy_item_id needs a deterministic uniqueness lookup (pluggy_items.pluggy_item_id_hash).
// NEVER substitute createHash('sha256') -- bare SHA-256 of a Pluggy itemId string is trivially pre-imageable.
export function hashPluggyItemId(plaintext: string): Buffer {
  const pepper = env.PLUGGY_ITEM_ID_HASH_PEPPER;
  if (!pepper) {
    // Sandbox/dev path: env.ts marks this optional outside production; fall back to a documented stub
    // to keep tests booting. Production refine in plan 02-01 prevents this branch in deployed builds.
    throw new Error('PLUGGY_ITEM_ID_HASH_PEPPER missing -- required for hashPluggyItemId()');
  }
  return createHmac('sha256', Buffer.from(pepper, 'utf8')).update(plaintext).digest();
}
```
Place the function next to `hashCPF` so callers can reach for either symbol from the same import path. Do NOT modify `encryptCPF`, `decryptCPF`, or `hashCPF` -- those keep their Phase 1 signatures.

import * as Sentry from '@sentry/nextjs';
import { PluggyClient } from 'pluggy-sdk';
import { decryptCPF as decrypt } from '@/lib/crypto';
import { logger } from '@/lib/logger';
import { hashUserIdForSentry as hashId } from '@/lib/sentry';
import { getPluggyClientId, getPluggyClientSecret } from '@/lib/pluggyEnv';

let _client: PluggyClient | null = null;

function getClient(): PluggyClient {
  if (_client) return _client;
  _client = new PluggyClient({
    clientId: getPluggyClientId(),
    clientSecret: getPluggyClientSecret(),
  });
  return _client;
}

export class PluggyError extends Error {
  // Wrap SDK errors so .message NEVER carries plaintext item id
  constructor(message: string, public readonly sdk_status?: number) {
    super(message);
    this.name = 'PluggyError';
  }
}

export class PluggyService {
  async createConnectToken(args: { user_id: string; reconnect_item_id_enc?: Buffer }): Promise<{ connect_token: string; expires_at: string }> {
    return Sentry.startSpan(
      { op: 'pluggy.create_connect_token', name: 'PluggyService.createConnectToken' },
      async () => {
        const item_id = args.reconnect_item_id_enc ? decrypt(args.reconnect_item_id_enc) : undefined;
        try {
          const t = await getClient().createConnectToken(item_id);    // SDK method name confirmed in 02-01-SUMMARY
          logger.info({
            event: 'pluggy_connect_token_created',
            user_id_hashed: hashId(args.user_id),
            reconnect: !!item_id,
          }, 'pluggy connect token created');
          return { connect_token: t.accessToken, expires_at: t.expiresAt };
        } catch (err) {
          // Strip any plaintext itemId from error message before re-throw
          throw new PluggyError(scrub_plaintext(String(err), item_id), (err as { status?: number }).status);
        }
      },
    );
  }

  async fetchItem(args: { user_id: string; item_id_enc: Buffer }) { /* ... same shape ... */ }
  async fetchAccounts(args: { user_id: string; item_id_enc: Buffer }) { /* ... */ }
  async fetchTransactions(args: { user_id: string; item_id_enc: Buffer; account_id: string; from: Date; to: Date; cursor?: string }) { /* ... */ }
  async deleteItem(args: { user_id: string; item_id_enc: Buffer }): Promise<void> {
    return Sentry.startSpan(
      { op: 'pluggy.delete_item', name: 'PluggyService.deleteItem' },
      async () => {
        const item_id = decrypt(args.item_id_enc);
        try {
          await getClient().deleteItem(item_id);
          logger.info({
            event: 'pluggy_item_deleted',
            user_id_hashed: hashId(args.user_id),
            item_id_hashed: hashId(item_id),
          }, 'pluggy item deleted');
        } catch (err) {
          throw new PluggyError(scrub_plaintext(String(err), item_id), (err as { status?: number }).status);
        }
      },
    );
  }
}

function scrub_plaintext(message: string, plaintext: string | undefined): string {
  if (!plaintext) return message;
  return message.split(plaintext).join('[redacted-pluggy-item-id]');
}

let _service: PluggyService | null = null;
export function getPluggyService(): PluggyService {
  if (_service) return _service;
  _service = new PluggyService();
  return _service;
}
