```typescript
export const runtime = 'nodejs';

import { NextResponse } from 'next/server';
import { z } from 'zod';
import { db } from '@/db';
import { users, user_consents } from '@/db/schema';
import { eq } from 'drizzle-orm';
import { requireSession } from '@/lib/session';
import { CPFSchema, encryptAndHashCPF } from '@/lib/cpf';
import { getConsentVersionHash } from '@/lib/consentVersions';
import { getPluggyService } from '@/services/PluggyService';
import { logger } from '@/lib/logger';

const Body = z.object({
  cpf: z.string().optional(),
  granted: z.literal(true),
  reconnect_item_id: z.string().uuid().optional(),
});

export async function POST(req: Request): Promise<Response> {
  const session = await requireSession(req);    // throws → 401
  let body: z.infer<typeof Body>;
  try { body = Body.parse(await req.json()); } catch { return NextResponse.json({ error: 'BAD_REQUEST' }, { status: 400 }); }

  // 1. CPF requirement: only when users.cpf_hash IS NULL
  const u = await db.select({ cpf_hash: users.cpf_hash }).from(users).where(eq(users.id, session.userId)).limit(1);
  const has_cpf = !!u[0]?.cpf_hash;
  if (!has_cpf) {
    if (!body.cpf) return NextResponse.json({ error: 'CPF_REQUIRED' }, { status: 400 });
    const parsed = CPFSchema.safeParse(body.cpf);
    if (!parsed.success) return NextResponse.json({ error: 'INVALID_CPF' }, { status: 400 });
    const { cpf_hash, cpf_enc } = encryptAndHashCPF(parsed.data);
    await db.update(users).set({ cpf_hash, cpf_enc }).where(eq(users.id, session.userId));
  }

  // 2. Pre-widget consent row (D-08 step 1)
  await db.insert(user_consents).values({
    user_id: session.userId,
    scope: 'PLUGGY_CONNECT_PENDING',
    action: 'GRANTED',
    consent_version_hash: getConsentVersionHash('pluggy_connect_v1'),
    ip_address: req.headers.get('x-forwarded-for'),
    user_agent: req.headers.get('user-agent'),
  });

  // 3. Issue Pluggy token
  let reconnect_item_id_enc: Buffer | undefined;
  if (body.reconnect_item_id) {
    const item = await db.query.pluggy_items.findFirst({
      where: (pi, { and, eq }) => and(eq(pi.id, body.reconnect_item_id!), eq(pi.user_id, session.userId)),
      columns: { pluggy_item_id_enc: true },
    });
    if (!item) return NextResponse.json({ error: 'NOT_FOUND' }, { status: 404 });    // P26: 404 not 403
    reconnect_item_id_enc = item.pluggy_item_id_enc;
  }
  const { connect_token, expires_at } = await getPluggyService().createConnectToken({
    user_id: session.userId,
    reconnect_item_id_enc,
  });
  return NextResponse.json({ connect_token, expires_at });
}
```

const Body = z.object({
  pluggy_item_id: z.string().min(1),
  connector_id: z.string().min(1),
  institution_name: z.string().min(1),
  institution_logo_url: z.string().url().optional().nullable(),
});

export async function POST(req: Request): Promise<Response> {
  const session = await requireSession(req);
  let body: z.infer<typeof Body>;
  try { body = Body.parse(await req.json()); } catch { return NextResponse.json({ error: 'BAD_REQUEST' }, { status: 400 }); }

  const item_enc = encrypt(body.pluggy_item_id);
  const item_hash = hashPluggyItemId(body.pluggy_item_id);   // RESEARCH OQ#6 RESOLVED -- HMAC-SHA-256 with PLUGGY_ITEM_ID_HASH_PEPPER, NOT bare createHash('sha256')

  let inserted_id: string;
  try {
    const rows = await db.insert(pluggy_items).values({
      user_id: session.userId,
      pluggy_item_id_enc: item_enc,
      pluggy_item_id_hash: item_hash,
      connector_id: body.connector_id,
      institution_name: body.institution_name,
      institution_logo_url: body.institution_logo_url ?? null,
      status: 'UPDATING',
    }).returning({ id: pluggy_items.id });
    inserted_id = rows[0].id;
  } catch (err) {
    // UNIQUE(user_id, pluggy_item_id_hash) violation
    if (String(err).includes('pluggy_items_user_item_hash_unique')) {
      return NextResponse.json({ error: 'ALREADY_CONNECTED' }, { status: 409 });
    }
    throw err;
  }

  // D-08 step 2: per-connector consent row
  await db.insert(user_consents).values({
    user_id: session.userId,
    scope: `PLUGGY_CONNECTOR:${body.connector_id}`,
    action: 'GRANTED',
    consent_version_hash: getConsentVersionHash('pluggy_connect_v1'),
    ip_address: req.headers.get('x-forwarded-for'),
    user_agent: req.headers.get('user-agent'),
  });

  // D-41: per-user singletonKey enqueue
  await enqueue(
    QUEUES.PLUGGY_SYNC,
    { user_id: session.userId, item_id: inserted_id, trigger: 'first_connect' },
    { singletonKey: session.userId, singletonHours: 0 },
  );

  await recordAudit({
    user_id: session.userId,
    action: 'item_connected',
    metadata: { connector_id: body.connector_id, institution_name: body.institution_name, cooldown_bypassed: false },
    ip_address: req.headers.get('x-forwarded-for'),
    user_agent: req.headers.get('user-agent'),
  });

  return NextResponse.json({ id: inserted_id }, { status: 202 });
}

export const runtime = 'nodejs';
import { NextResponse } from 'next/server';
import { db } from '@/db';
import { pluggy_items, accounts, transactions } from '@/db/schema';
import { eq, and, sql } from 'drizzle-orm';
import { requireSession } from '@/lib/session';

export async function GET(req: Request): Promise<Response> {
  const session = await requireSession(req);
  // Find user's most recent pluggy_item
  const item = await db.query.pluggy_items.findFirst({
    where: (pi, { eq }) => eq(pi.user_id, session.userId),
    orderBy: (pi, { desc }) => [desc(pi.created_at)],
    columns: { id: true, status: true },
  });
  if (!item) return NextResponse.json({ phase: 'no_items', transactions_count: 0 });

  const accountCount = await db.select({ n: sql`count(*)::int` }).from(accounts).where(and(eq(accounts.pluggy_item_id, item.id)));
  const txCount = await db.select({ n: sql`count(*)::int` }).from(transactions).where(eq(transactions.user_id, session.userId));
  const tx_n = (txCount[0]?.n as number) ?? 0;

  let phase: 'connecting' | 'loading_accounts' | 'loading_transactions' | 'completed';
  if (tx_n > 0) phase = 'completed';
  else if ((accountCount[0]?.n as number) > 0) phase = 'loading_transactions';
  else if (item.status === 'UPDATING') phase = 'loading_accounts';
  else phase = 'connecting';
  return NextResponse.json({ phase, transactions_count: tx_n });
}
